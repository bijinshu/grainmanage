Demo1(一个基础的配置)

$("#demo1").myPagination({});

Demo2(更改样式)

$("#demo2").myPagination({cssStyle:'scott'});

Demo3(初始化页面及更改页码大小)

$("#demo3").myPagination({cssStyle:"black",currPage:3,pageCount:100});

Demo4(配置页码栏,最后一页不显示)

$("#demo4").myPagination({cssStyle:'starcraft2',panel:{first: '<<',last: '>>', next: '>',prev: '<',links:"javascript:void(0);",last_on:false}});

Demo5(配置文本框样式)

$("#demo5").myPagination({cssStyle:'grayr',panel:{tipInfo_on:true,tipInfo_css:{width:"22px",height:'12px'}}});

Demo6(配置onClick提示信息)

$("#demo6").myPagination({cssStyle:'jogger',ajax:{on:false,onClick:function(page){ ZENG.msgbox.show(" 正在加载"+page+"页，请稍后...", 6, 1000);}}});
