动态实例演示(需在Web环境下运行)

Demo1(Json) 

        jsonMyPagination = $("#demo1").myPagination({
          ajax: {
            on: true,
            url: "jsonData.txt",
            dataType: 'json',
            ajaxStart: function() {
              ZENG.msgbox.show(" 正在加载中，请稍后...", 6, 10000);
            },
            callback: function(data) {
              //用于测试，看到提示效果。
              setTimeout(function() {
                ZENG.msgbox.hide(); //隐藏加载提示
              }, 500);
              $("#demo1_result").html("ajax请求结果：" + objToString(data));
            }
          }
        });

Demo2(Xml) 

        xmlMyPagination = $("#demo2").myPagination({
          cssStyle: "green-black",
          ajax: {
            on: true,
            url: "xmlData.xml",
            dataType: 'xml',
            ajaxStart: function() {
              ZENG.msgbox.show(" 正在加载中，请稍后...", 6, 10000);
            },
            callback: function(data) {
              //用于测试，看到提示效果。
              setTimeout(function() {
                ZENG.msgbox.hide(); //隐藏加载提示
              }, 500);
			  
			  var result = "";
			  $(data).find("book").each(function(i) {
				var id = $(this).attr("id");
				var name = $(this).find("name").text();
				var author = $(this).find("author").text();
				var price = $(this).find("price").text();
				result += "book:("+id+","+name+","+author+","+price+")";
			  });
              $("#demo2_result").html("ajax请求结果：" + result);
            }
          }
        });
